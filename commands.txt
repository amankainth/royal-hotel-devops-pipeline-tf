Jenkin fix on Fiserv Network

# Stop Jenkins
sudo systemctl stop jenkins

# Clean up old override
sudo rm -f /etc/systemd/system/jenkins.service.d/override.conf

# Create proper systemd override (uses AmbientCapabilities - safe method)
sudo mkdir -p /etc/systemd/system/jenkins.service.d
sudo tee /etc/systemd/system/jenkins.service.d/override.conf > /dev/null <<'EOF'
[Service]
Environment="JAVA_HOME=/usr/lib/jvm/java-21-amazon-corretto"
AmbientCapabilities=CAP_NET_BIND_SERVICE
CapabilityBoundingSet=CAP_NET_BIND_SERVICE
ExecStart=
ExecStart=/usr/bin/java -Djava.awt.headless=true -jar /usr/share/java/jenkins.war --webroot=/var/cache/jenkins/war --httpPort=80
EOF

# Reload and start
sudo systemctl daemon-reload
sudo systemctl reset-failed jenkins
sudo systemctl start jenkins

# Wait for startup
echo "Waiting for Jenkins to start on port 80..."
sleep 20

# Show results
echo ""
echo "=== Jenkins Status ==="
sudo systemctl status jenkins --no-pager | head -10
echo ""
echo "=== Port 80 Listener ==="
sudo ss -tlnp | grep ':80 '
echo ""
echo "=== Local Test ==="
curl -sI http://localhost 2>&1 | head -3
echo ""
echo "=== Access URL ==="
echo "http://$(curl -s http://checkip.amazonaws.com)"
